473 Vector3f Matrixlib::eigenvaluesSym(const Matrix3f &m)
479 {
480  assert( ((m - m.transpose())).squaredNorm() < EPSILON );
481  Vector3f w;
482  float s1,s2,s3;
483  // check if m is diagonal;
484  if (m(0,1)*m(0,1)+m(0,2)*m(0,2)+m(1,2)*m(1,2)<EPSILON){
485  s1 = m(0,0);
486  s2 = m(1,1);
487  s3 = m(2,2);
488  }else{
489  float a = - m.trace();
490  float aa = a*a;
491  float b = (aa - m.squaredNorm()) / 2;
492 
493  float p = aa / 3 - b;
494  float q = 2 * aa * a / 27 - a * b / 3 - m.determinant();
495 
496  // since p,q are close to 0, the next statement is numerically unstable.
497  if (p==0) {
498  s1 = s2 = s3 = -a/3;
499  }else {
500  float r = sqrtf(4 * p / 3);
501  float k = - 4 * q / (r * r * r);
502  float theta;
503  if (k>1.0){
504  theta = 0.0;
505  }else if(k<-1.0){
506  theta = M_PI;
507  }else{
508  theta = acosf(k);
509  }
510  s1 = r*cosf(theta / 3) - a / 3;
511  s2 = r*cosf((theta + 2 * M_PI) / 3) - a / 3;
512  s3 = r*cosf((theta + 4 * M_PI) / 3) - a / 3;
513  }
514  }
515  // sort
516  if (s1 < s2) {
517  if (s3 < s1) std::swap(s1,s3);
518  } else {
519  if (s2 < s3) std::swap(s1,s2);
520  else std::swap(s1,s3);
521  }
522  if(s3<s2) std::swap(s2,s3);
523  w << s1, s2, s3;
524  return w;
